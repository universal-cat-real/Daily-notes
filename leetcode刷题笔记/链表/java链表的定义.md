```java
public class ListNode {  
  
    //结点的值  
    int val;  
    //下一个结点  
    ListNode next;  
  
    public ListNode() {  
  
    }  
    public ListNode(int val){  
        this.val = val;  
    }  
    public ListNode(int val, ListNode next){  
        this.val = val;  
        this.next = next;  
    }  
}
```
